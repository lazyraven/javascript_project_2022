Count vowels in a string

Given a string s, return the total count of vowel characters (a, e, i, o, u) — both uppercase and lowercase. Use a frequency counter or set to check membership efficiently.

INPUT
s = "Hello World"
OUTPUT
3  // e, o, o
INPUT
s = "JavaScript"
OUTPUT
3  // a, a, i
INPUT
s = "rhythm"
OUTPUT
0
💡 Hint: Use a Set of vowels for O(1) lookup. Iterate characters, increment counter if vowel.
solution.js
function countVowels(s) {
  const vowels = new Set(['a','e','i','o','u']);
  let count = 0;
  for (const ch of s.toLowerCase()) {
    if (vowels.has(ch)) count++;
  }
  return count;
}
// Time: O(n) | Space: O(1)
Count characters in a string

Given a string s, return an object (frequency map) where each key is a character and the value is how many times it appears. Ignore spaces.

INPUT
s = "hello"
OUTPUT
{ h:1, e:1, l:2, o:1 }
INPUT
s = "aabbcc"
OUTPUT
{ a:2, b:2, c:2 }
INPUT
s = "aab"
OUTPUT
{ a:2, b:1 }
💡 Hint: Classic frequency counter. Loop, build an object. This is the core pattern used everywhere.
solution.js
function charCount(s) {
  const freq = {};
  for (const ch of s) {
    if (ch !== ' ')
      freq[ch] = (freq[ch] || 0) + 1;
  }
  return freq;
}
// Time: O(n) | Space: O(k) — k = unique chars
Find duplicates in array

Given an array of integers nums, return all elements that appear more than once. The result may be in any order.

INPUT
nums = [1,2,3,2,4,3,5]
OUTPUT
[2, 3]
INPUT
nums = [4,4,4,1,2]
OUTPUT
[4]
INPUT
nums = [1,2,3]
OUTPUT
[]
💡 Hint: Build frequency map first. Then filter keys where count > 1.
solution.js
function findDuplicates(nums) {
  const freq = {};
  for (const n of nums)
    freq[n] = (freq[n] || 0) + 1;
  return Object.keys(freq).filter(k => freq[k] > 1).map(Number);
}
// Time: O(n) | Space: O(n)
Find unique elements

Given an array, return only the elements that appear exactly once. Order doesn't matter.

INPUT
nums = [1,2,2,3,4,4,5]
OUTPUT
[1, 3, 5]
INPUT
nums = [7,7,7]
OUTPUT
[]
INPUT
nums = [9]
OUTPUT
[9]
💡 Hint: Same as duplicates — just filter where count === 1 instead of > 1.
solution.js
function findUnique(nums) {
  const freq = {};
  for (const n of nums)
    freq[n] = (freq[n] || 0) + 1;
  return Object.keys(freq)
    .filter(k => freq[k] === 1)
    .map(Number);
}
// Time: O(n) | Space: O(n)
First non-repeating character

Given a string s, find the first character that appears exactly once and return its index. If none exist, return -1. (LeetCode #387)

INPUT
s = "leetcode"
OUTPUT
0  // 'l' appears once
INPUT
s = "loveleetcode"
OUTPUT
2  // 'v'
INPUT
s = "aabb"
OUTPUT
-1
💡 Hint: Two passes — first build frequency map, then scan again to find first char with count 1.
solution.js
function firstUniqChar(s) {
  const freq = {};
  for (const ch of s)
    freq[ch] = (freq[ch] || 0) + 1;
  for (let i = 0; i < s.length; i++) {
    if (freq[s[i]] === 1) return i;
  }
  return -1;
}
// Time: O(n) | Space: O(1) — at most 26 keys
● INTERMEDIATE
Valid Anagram

Given two strings s and t, return true if t is an anagram of s, and false otherwise. An anagram uses all the same characters the same number of times. (LeetCode #242)

INPUT
s = "anagram"
t = "nagaram"
OUTPUT
true
INPUT
s = "rat"
t = "car"
OUTPUT
false
INPUT
s = "ab"
t = "a"
OUTPUT
false  // different lengths
💡 Hint: Build freq map for s, then decrement using t. If all values are 0 → anagram.
solution.js
function isAnagram(s, t) {
  if (s.length !== t.length) return false;
  const freq = {};
  for (const ch of s)
    freq[ch] = (freq[ch] || 0) + 1;
  for (const ch of t) {
    if (!freq[ch]) return false;
    freq[ch]--;
  }
  return true;
}
// Time: O(n) | Space: O(1)
Group Anagrams

Given an array of strings strs, group the anagrams together. Return the groups in any order. (LeetCode #49)

INPUT
strs = ["eat","tea","tan","ate","nat","bat"]
OUTPUT
[["eat","tea","ate"],
 ["tan","nat"],
 ["bat"]]
INPUT
strs = [""]
OUTPUT
[[""]]
💡 Hint: Sort each word as a key — anagrams have the same sorted form. Use a Map to group them.
solution.js
function groupAnagrams(strs) {
  const map = new Map();
  for (const s of strs) {
    const key = s.split('').sort().join('');
    if (!map.has(key)) map.set(key, []);
    map.get(key).push(s);
  }
  return [...map.values()];
}
// Time: O(n·k·log k) | Space: O(n·k)
Majority Element

Given array nums of size n, return the element that appears more than ⌊n/2⌋ times. It is guaranteed the majority element always exists. (LeetCode #169)

INPUT
nums = [3,2,3]
OUTPUT
3
INPUT
nums = [2,2,1,1,1,2,2]
OUTPUT
2
INPUT
nums = [6,5,5]
OUTPUT
5
💡 Bonus: Boyer-Moore Voting Algorithm solves this in O(1) space. For interviews, frequency counter is clearer.
solution.js
function majorityElement(nums) {
  const freq = {};
  const half = nums.length / 2;
  for (const n of nums) {
    freq[n] = (freq[n] || 0) + 1;
    if (freq[n] > half) return +n;
  }
}
// Time: O(n) | Space: O(n)
Top K Frequent Elements

Given an integer array nums and an integer k, return the k most frequently occurring elements. Answer may be in any order. (LeetCode #347)

INPUT
nums = [1,1,1,2,2,3], k = 2
OUTPUT
[1, 2]
INPUT
nums = [1], k = 1
OUTPUT
[1]
INPUT
nums = [4,1,1,2,2,2], k = 2
OUTPUT
[2, 1]
💡 Hint: Build freq map → sort keys by frequency descending → slice first k elements.
solution.js
function topKFrequent(nums, k) {
  const freq = {};
  for (const n of nums)
    freq[n] = (freq[n] || 0) + 1;
  return Object.keys(freq)
    .sort((a,b) => freq[b] - freq[a])
    .slice(0, k)
    .map(Number);
}
// Time: O(n log n) | Space: O(n)
Frequency Sort (Sort by Character Frequency)

Given a string s, sort it in decreasing order based on the frequency of characters. If two characters have the same frequency, their order can be arbitrary. (LeetCode #451)

INPUT
s = "tree"
OUTPUT
"eert"  // e appears twice
INPUT
s = "cccaaa"
OUTPUT
"aaaccc" or "cccaaa"
INPUT
s = "Aabb"
OUTPUT
"bbAa" or "bbaA"
💡 Hint: Build freq map → sort characters by freq desc → repeat each character freq times.
solution.js
function frequencySort(s) {
  const freq = {};
  for (const ch of s)
    freq[ch] = (freq[ch] || 0) + 1;
  return Object.keys(freq)
    .sort((a,b) => freq[b] - freq[a])
    .map(ch => ch.repeat(freq[ch]))
    .join('');
}
// Time: O(n log n) | Space: O(n)
● ADVANCED
Sliding Window + Frequency Counter

Given a string s and a string p, find all start indices of p's anagrams in s. Return indices in any order. (LeetCode #438 — Find All Anagrams in a String)

INPUT
s = "cbaebabacd"
p = "abc"
OUTPUT
[0, 6]
// "cba" at 0, "bac" at 6
INPUT
s = "abab"
p = "ab"
OUTPUT
[0, 1, 2]
💡 Hint: Maintain a sliding window of size p.length. Keep two freq maps — slide by adding right char and removing left char. Compare maps.
solution.js
function findAnagrams(s, p) {
  const result = [];
  const pFreq = {}, wFreq = {};
  for (const ch of p)
    pFreq[ch] = (pFreq[ch] || 0) + 1;

  let matches = 0;
  const required = Object.keys(pFreq).length;

  for (let r = 0; r < s.length; r++) {
    const ch = s[r];
    wFreq[ch] = (wFreq[ch] || 0) + 1;
    if (pFreq[ch] && wFreq[ch] === pFreq[ch]) matches++;

    if (r >= p.length) {
      const left = s[r - p.length];
      if (pFreq[left] && wFreq[left] === pFreq[left]) matches--;
      wFreq[left]--;
    }

    if (matches === required) result.push(r - p.length + 1);
  }
  return result;
}
// Time: O(n) | Space: O(1)
Minimum Window Substring

Given strings s and t, return the minimum window in s which contains all characters of t. If no such window exists, return "". (LeetCode #76 — Hard)

INPUT
s = "ADOBECODEBANC"
t = "ABC"
OUTPUT
"BANC"
INPUT
s = "a", t = "a"
OUTPUT
"a"
INPUT
s = "a", t = "aa"
OUTPUT
""  // not enough 'a'
💡 Hint: Expand window right until all t chars are covered, then shrink from left. Track how many unique chars are "satisfied".
solution.js
function minWindow(s, t) {
  const need = {}, have = {};
  for (const ch of t) need[ch] = (need[ch] || 0) + 1;
  let formed = 0, required = Object.keys(need).length;
  let l = 0, res = "", minLen = Infinity;
  for (let r = 0; r < s.length; r++) {
    const ch = s[r];
    have[ch] = (have[ch] || 0) + 1;
    if (need[ch] && have[ch] === need[ch]) formed++;
    while (formed === required) {
      if (r - l + 1 < minLen) { minLen = r - l + 1; res = s.slice(l, r+1); }
      const lc = s[l++];
      have[lc]--;
      if (need[lc] && have[lc] < need[lc]) formed--;
    }
  }
  return res;
}
// Time: O(n) | Space: O(k)
Longest Substring Without Repeating Characters

Given a string s, find the length of the longest substring without repeating characters. (LeetCode #3)

INPUT
s = "abcabcbb"
OUTPUT
3  // "abc"
INPUT
s = "bbbbb"
OUTPUT
1  // "b"
INPUT
s = "pwwkew"
OUTPUT
3  // "wke"
💡 Hint: Use a Set and two pointers. When a duplicate is found, shrink from left. Max window = answer.
solution.js
function lengthOfLongestSubstring(s) {
  const set = new Set();
  let l = 0, max = 0;
  for (let r = 0; r < s.length; r++) {
    while (set.has(s[r])) set.delete(s[l++]);
    set.add(s[r]);
    max = Math.max(max, r - l + 1);
  }
  return max;
}
// Time: O(n) | Space: O(min(n, m))

Real Interview Insight

Frequency Counter pattern appears everywhere in:

  • Array problems — duplicates, missing numbers, majority elements
  • String problems — anagrams, character counts, palindrome checks
  • HashMap problems — two-sum variants, subarray sums
  • Sliding window problems — substring problems, max window conditions
  • Data analysis tasks — top-k, ranking, histograms

If you master frequency counters, many medium-level problems become straightforward.

// Core Frequency Counter Template
function countOccurrences(arr) {
  const frequency = {};

  for (const item of arr) {
    frequency[item] = (frequency[item] || 0) + 1;
  }

  return frequency;
}

// With Map (recommended for non-string keys)
function countWithMap(arr) {
  const freq = new Map();
  for (const item of arr)
    freq.set(item, (freq.get(item) || 0) + 1);
  return freq;
}
