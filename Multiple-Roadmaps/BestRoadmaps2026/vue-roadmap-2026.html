Interview-Level Code
💡 Todo App — What Interviews Test
TodoApp.vue — tests: ref, computed, watch, v-model 
<script setup>
import { ref, computed, watch } from "vue";

const todos = ref(
  JSON.parse(localStorage.getItem("todos")) || []
); // ref for a reactive array

const input = ref("");

const remaining = computed(() =>
  todos.value.filter(t => !t.done).length
); // derived, cached state

watch(todos, (val) => {
  localStorage.setItem("todos", JSON.stringify(val));
}, { deep: true }); // runs whenever todos changes

function addTodo() {
  if (!input.value.trim()) return;
  todos.value.push({ id: Date.now(), text: input.value, done: false });
  input.value = "";
}

function toggleTodo(id) {
  const t = todos.value.find(t => t.id === id);
  if (t) t.done = !t.done;
}
</script>

<template>
  <div>
    <input v-model="input" @keyup.enter="addTodo" />
    <button @click="addTodo">Add</button>
    <p>{{ remaining }} left</p>
    <div v-for="todo in todos" :key="todo.id" @click="toggleTodo(todo.id)">
      {{ todo.text }} {{ todo.done ? "✔" : "" }}
    </div>
  </div>
</template>
